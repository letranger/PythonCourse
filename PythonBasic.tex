\label{sec:org31b395d}
\begin{enumerate}
\item 啟動方式
\label{sec:orgcec1810}
\lstset{breaklines=true,language=shell,label= ,caption= ,captionpos=b,numbers=none}
\begin{lstlisting}
python -i filename.py
\end{lstlisting}
\item 優點\footnote{\href{https://stackabuse.com/python-programming-in-interactive-vs-script-mode/}{Python Programming in Interactive vs Script Mode}}
\label{sec:org1ab8708}
\begin{itemize}
\item Helpful when your script is extremely short and you want immediate results.
\item Faster as you only have to type a command and then press the enter key to get the results.
\item Good for beginners who need to understand Python basics.
\end{itemize}
\item 缺點
\label{sec:org3a19552}
\begin{itemize}
\item Editing the code in interactive mode is hard as you have to move back to the previous commands or else you have to rewrite the whole command again.
\item It's very tedious to run long pieces of code.
\end{itemize}
\end{enumerate}
\subsubsection{Script Mode}
\label{sec:org920d168}
\begin{enumerate}
\item 啟動方式
\label{sec:orgd878c90}
\lstset{breaklines=true,language=bash,label= ,caption= ,captionpos=b,numbers=none}
\begin{lstlisting}
python filename.py
\end{lstlisting}
\item 優點:
\label{sec:orgec8ea72}
\begin{itemize}
\item It is easy to run large pieces of code.
\item Editing your script is easier in script mode.
\item Good for both beginners and experts.
\end{itemize}
\item 缺點:
\label{sec:org2c7ea73}
\begin{itemize}
\item Can be tedious when you need to run only a single or a few lines of cod.
\item You must create and save a file before executing your code.
\end{itemize}
\end{enumerate}

\subsubsection{格式化輸出}
\label{sec:org0c0cad9}
\begin{itemize}
\item 語法
\begin{itemize}
\item 格式化輸出語法：print( 字串  \%(參數) ) 
\item 字串裡 \%s 代表字串、\%d 代表整數、\%f 代表浮點數
\end{itemize}

\item 範例
\lstset{breaklines=true,language=Python,label= ,caption= ,captionpos=b,firstnumber=1,numbers=left}
\begin{lstlisting}
name = '台北101'
height = 508
fee = 18.24
print('%s的高度為%d公尺，參觀門票為%3.2f美金.' %(name, height, fee))

\end{lstlisting}

\begin{verbatim}
台北101的高度為508公尺，參觀門票為18.24美金.
\end{verbatim}
\end{itemize}
\subsubsection{format格式化輸出}
\label{sec:org5bf39f4}
\begin{itemize}
\item 語法 
\begin{itemize}
\item format格式化輸出語法：print(字串.format (參數) ) 
\item 字串裡以\{0\}、\{1\}、\ldots{} 來對應參數列裡的變數
\end{itemize}

\item 範例
\lstset{breaklines=true,language=Python,label= ,caption= ,captionpos=b,firstnumber=1,numbers=left}
\begin{lstlisting}
name = 'Taipei101'
height = 508.35
print('{0}的高度為{1}公尺'.format (name, height))
print('{0:15s}的高度為{1:10.1f}公尺'.format (name, height))
\end{lstlisting}

\begin{verbatim}
Taipei101的高度為508.35公尺
Taipei101      的高度為     508.4公尺
\end{verbatim}
\end{itemize}

\subsubsection{表列函式與方法}
\label{sec:org18dd410}
\begin{itemize}
\item len(<list>): 計算表列元素個數
\item list(<str>): 將<str>轉成表列
\item <list>.clear():	清除表列中所有元素
\item <list>.append(<obj>): 將<obj>加到<list>尾端
\item <list>.extend(<list1>): 將<list1>合併至<list>尾端
\item <list>.remove(<obj>): 移除<list>中<obj>元素
\item <list>.insert(<i>,<obj>): 將<obj>加到<list>的索引值<i>位置

\subsection{遞迴(recursive)}
\label{sec:orgd0a0c0b}
\subsubsection{語法}
\label{sec:org760368c}
\begin{itemize}
\item 遞迴(recursive)：一個函式直接呼叫函式本身
\end{itemize}
\lstset{breaklines=true,language=Python,label= ,caption= ,captionpos=b,firstnumber=1,numbers=left}
\begin{lstlisting}
def fib(n):
    if n == 1 or n == 2:
        return 1
    else:
        return fib(n-1) + fib(n-2)

print('The %d-th Fibonacci number = %d' %(10, fib(10)))
\end{lstlisting}

\begin{verbatim}
The 10-th Fibonacci number = 55
\end{verbatim}

\subsubsection{基本條件}
\label{sec:orge6c9d22}
\begin{itemize}
\item 遞迴程式設計要滿足兩條件:
\begin{enumerate}
\item 遞迴關係式，如 
\lstset{breaklines=true,language=Python,label= ,caption= ,captionpos=b,firstnumber=1,numbers=left}
\begin{lstlisting}
fib(n) = fib(n-1) + fib(n-2)
sum(n) = n + sum(n-1) 
\end{lstlisting}

\item 遞迴終止式，如
\lstset{breaklines=true,language=Python,label= ,caption= ,captionpos=b,firstnumber=1,numbers=left}
\begin{lstlisting}
f(1) = f(2) = 1
sum(0) = 0
\end{lstlisting}
\end{enumerate}

\item 範例:GCD-while
\lstset{breaklines=true,language=Python,label= ,caption= ,captionpos=b,firstnumber=1,numbers=left}
\begin{lstlisting}
m, n = 20, 50
while (n>0):
    m, n = n, m % n
print(m)
\end{lstlisting}

\begin{verbatim}

\end{verbatim}
\end{itemize}

\subsubsection{練習}

\begin{itemize}
\item 計算 n! 值。 輸入 n，輸出 n! = XXXXX。如輸入5，輸出5! = 120。
\item 計算C(n, r)組合數。 輸入 n, r，輸出C(n, r) = XX。如輸入9, 4，輸出C(9, 4) = 126。
\item 計算擴展歐幾里德。 輸入 a, b，輸出a*x + b*y = gcd。如輸入3, 5，輸出3*2 + 5*-1 = 1
\lstset{breaklines=true,language=Python,label= ,caption= ,captionpos=b,firstnumber=1,numbers=left}
\begin{lstlisting}
def ext_euclid(a, b):
    old_s,s=1,0
    old_t,t=0,1
    old_r,r=a,b
    if b == 0:
        return 1, 0, a
    else:
        while(r!=0):
            q=old_r//r
            old_r,r=r,old_r-q*r
            old_s,s=s,old_s-q*s
            old_t,t=t,old_t-q*t
    return old_s, old_t, old_r

a, b, c = ext_euclid(240, 33)
print('%dx + %dy = %d' %(a, b, c))
\end{lstlisting}

\begin{verbatim}
4x + -29y = 3
\end{verbatim}
\end{itemize}
