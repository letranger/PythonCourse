\subsection{特殊矩陣}

\subsection{矩陣形狀轉換}

\begin{itemize}
\item reshape(m, n, …)：不改變原矩陣形狀\\

import urllib.request as req

titles = root.find("div", class_="title")
print(titles)
print(titles.a.string)
titles = root.find_all("div", class_="title")
print(titles)
for title in titles:
    if title.a != None:
        print(title.a.string)
