* 統計錄影時間

from os import listdir
from os.path import join
from moviepy.editor import VideoFileClip

def caltime(path):
    files = listdir(path)

    tot = 0
    for f in files:
        if f.endswith("mp4"):
           fullpath = join(path, f)
           clip = VideoFileClip(fullpath)
           tot += clip.duration
    return tot/3600

path = "/Users/letranger/Desktop/研習/入門"
print("入門: %.2f小時" %(caltime(path)))
